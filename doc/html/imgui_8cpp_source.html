95  void closestPointOnLine(float& ox, float &oy, float px, float py, float ax, float ay, float bx, float by)
96  {
97  float dx = px - ax;
98  float dy = py - ay;
99 
100  float lx = bx - ax;
101  float ly = by - ay;
102 
103  float len = sqrtf(lx*lx+ly*ly);
104 
105  // Normalize.
106  float invLen = 1.0f/len;
107  lx*=invLen;
108  ly*=invLen;
109 
110  float dot = (dx*lx + dy*ly);
111 
112  if (dot < 0.0f)
113  {
114  ox = ax;
115  oy = ay;
116  }
117  else if (dot > len)
118  {
119  ox = bx;
120  oy = by;
121  }
122  else
123  {
124  ox = ax + lx*dot;
125  oy = ay + ly*dot;
