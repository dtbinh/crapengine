174  void frame(const void* _data)
175  {
176  if (NULL != m_frame)
177  {
178  ++m_numFrames;
179  uint32_t width = m_width;
180  uint32_t height = m_height;
181 
182  uint8_t* bgr = &m_frame[8];
183 
184  if (m_yflip)
185  {
186  for (uint32_t yy = 0; yy < height; ++yy)
187  {
188  const uint8_t* bgra = (const uint8_t*)_data + (height-1-yy)*width*4;
189 
190  for (uint32_t ii = 0; ii < width; ++ii)
191  {
192  bgr[0] = bgra[0];
193  bgr[1] = bgra[1];
194  bgr[2] = bgra[2];
195  bgr += 3;
196  bgra += 4;
197  }
198  }
199  }
200  else
201  {
202  const uint8_t* bgra = (const uint8_t*)_data;
203  for (uint32_t ii = 0, num = m_frameSize/3; ii < num; ++ii)
204  {
205  bgr[0] = bgra[0];
206  bgr[1] = bgra[1];
207  bgr[2] = bgra[2];
208  bgr += 3;
209  bgra += 4;
210  }
211  }
212 
213  bx::write(m_writer, m_frame, m_frameSize+8);
214  }
