107  void update(float _deltaTime, uint32_t _mx, uint32_t _my, bool _move)
108  {
109  if (!m_mouseDown)
110  {
111  m_mouseLast.m_mx = _mx;
112  m_mouseLast.m_my = _my;
113  }
114

116 
117  if (_move)
118  {
119  m_mouseNow.m_mx = _mx;
120  m_mouseNow.m_my = _my;
121  }
122 
123  if (m_mouseDown)
124  {
125  int32_t deltaX = m_mouseNow.m_mx - m_mouseLast.m_mx;
126  int32_t deltaY = m_mouseNow.m_my - m_mouseLast.m_my;
127 
128  m_horizontalAngle += m_mouseSpeed * float(deltaX);
129  m_verticalAngle -= m_mouseSpeed * float(deltaY);
130 
131  m_mouseLast.m_mx = m_mouseNow.m_mx;
132  m_mouseLast.m_my = m_mouseNow.m_my;
133  }

135  float direction[3] =
136  {
137  cosf(m_verticalAngle) * sinf(m_horizontalAngle),
138  sinf(m_verticalAngle),
139  cosf(m_verticalAngle) * cosf(m_horizontalAngle),
140  };
141 
142  float right[3] =
143  {
144  sinf(m_horizontalAngle - float(M_PI)/2.0f),

148
