688 void TextBufferManager::submitTextBuffer(TextBufferHandle _handle, uint8_t _id, int32_t _depth)
689 {
690  BX_CHECK(bgfx::isValid(_handle), "Invalid handle used");
691 
692  BufferCache& bc = m_textBuffers[_handle.idx];
693 
694  uint32_t indexSize = bc.textBuffer->getIndexCount() * bc.textBuffer->getIndexSize();
695  uint32_t vertexSize = bc.textBuffer->getVertexCount() * bc.textBuffer->getVertexSize();
696 
697  if (0 == indexSize || 0 == vertexSize)
698  {
699  return;
700  }
701 
702  bgfx::setTexture(0, u_texColor, m_fontManager->getAtlas()->getTextureHandle() );
703 
704  switch (bc.fontType)
705  {
706  case FONT_TYPE_ALPHA:
707  bgfx::setProgram(m_basicProgram);
708  bgfx::setState(0
709  | BGFX_STATE_RGB_WRITE
710  | BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA)
711  );
712  break;
713 
714  case FONT_TYPE_DISTANCE:
715  bgfx::setProgram(m_distanceProgram);
716  bgfx::setState(0
717  | BGFX_STATE_RGB_WRITE
718  | BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA)
719  );
720  break;
721 
722  case FONT_TYPE_DISTANCE_SUBPIXEL:
723  bgfx::setProgram(m_distanceSubpixelProgram);
724  bgfx::setState(0
725  | BGFX_STATE_RGB_WRITE
726  | BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_FACTOR, BGFX_STATE_BLEND_INV_SRC_COLOR)
727  , bc.textBuffer->getTextColor()
728  );
729  break;
730  }
731 
732  switch (bc.bufferType)
733  {
734  case BufferType::Static:
735  {
736  bgfx::IndexBufferHandle ibh;
737  bgfx::VertexBufferHandle vbh;
738 
739  if (bgfx::invalidHandle == bc.vertexBufferHandleIdx)
740  {
741  ibh = bgfx::createIndexBuffer(
742  bgfx::copy(bc.textBuffer->getIndexBuffer(), indexSize)
743  );
744 
745  vbh = bgfx::createVertexBuffer(
746  bgfx::copy(bc.textBuffer->getVertexBuffer(), vertexSize)
747  , m_vertexDecl
748  );
749 
750  bc.vertexBufferHandleIdx = vbh.idx;
751  bc.indexBufferHandleIdx = ibh.idx;
752  }
753  else
754  {
755  vbh.idx = bc.vertexBufferHandleIdx;
756  ibh.idx = bc.indexBufferHandleIdx;
757  }
758 
759  bgfx::setVertexBuffer(vbh, 0, bc.textBuffer->getVertexCount() );
760  bgfx::setIndexBuffer(ibh, bc.textBuffer->getIndexCount() );
761  }
762  break;
763 
764  case BufferType::Dynamic:
765  {
766  bgfx::DynamicIndexBufferHandle ibh;
767  bgfx::DynamicVertexBufferHandle vbh;
768 
769  if (bgfx::invalidHandle == bc.vertexBufferHandleIdx )
770  {
771  ibh = bgfx::createDynamicIndexBuffer(
772  bgfx::copy(bc.textBuffer->getIndexBuffer(), indexSize)
773  );
774 
775  vbh = bgfx::createDynamicVertexBuffer(
776  bgfx::copy(bc.textBuffer->getVertexBuffer(), vertexSize)
777  , m_vertexDecl
778  );
779 
780  bc.indexBufferHandleIdx = ibh.idx;
781  bc.vertexBufferHandleIdx = vbh.idx;
782  }
783  else
784  {
785  ibh.idx = bc.indexBufferHandleIdx;
786  vbh.idx = bc.vertexBufferHandleIdx;
787 
788  bgfx::updateDynamicIndexBuffer(ibh
789  , bgfx::copy(bc.textBuffer->getIndexBuffer(), indexSize)
790  );
791 
792  bgfx::updateDynamicVertexBuffer(vbh
793  , bgfx::copy(bc.textBuffer->getVertexBuffer(), vertexSize)
794  );
795  }
796 
797  bgfx::setVertexBuffer(vbh, bc.textBuffer->getVertexCount() );
798  bgfx::setIndexBuffer(ibh, bc.textBuffer->getIndexCount() );
799  }
800  break;
801 
802  case BufferType::Transient:
803  {
804  bgfx::TransientIndexBuffer tib;
805  bgfx::TransientVertexBuffer tvb;
806  bgfx::allocTransientIndexBuffer(&tib, bc.textBuffer->getIndexCount() );
807  bgfx::allocTransientVertexBuffer(&tvb, bc.textBuffer->getVertexCount(), m_vertexDecl);
808  memcpy(tib.data, bc.textBuffer->getIndexBuffer(), indexSize);
809  memcpy(tvb.data, bc.textBuffer->getVertexBuffer(), vertexSize);
810  bgfx::setVertexBuffer(&tvb, 0, bc.textBuffer->getVertexCount() );
811  bgfx::setIndexBuffer(&tib, 0, bc.textBuffer->getIndexCount() );
812  }
813  break;
814  }
815 
816  bgfx::submit(_id, _depth);
817 }
