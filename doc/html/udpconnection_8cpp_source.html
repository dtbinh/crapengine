158 bool UdpConnection::update( uint32_t deltatime )
159 {
160  _passed_time += deltatime;
161  if( _passed_time >= _update_frequency )
162  {
163  CRAP_DEBUG_LOG( LOG_NETWORK, "[CON] Updating %u connections", _connections.size() );
164  for( uint32_t i = 0; i< _connections.size(); ++i )
165  {
166  const uint32_t current_time = _connections.get_value(i)->user_time;
167  const uint32_t new_time = current_time + deltatime;
168 
169  if( new_time >= _idle_time )
170  {
171  sendLogout( _connections.get_value(i)->user_ip, _connections.get_value(i)->user_port );
172  _connections.erase_at(i--);
173  continue;
174  }
175 
176  sendSync( _connections.get_value(i)->user_ip, _connections.get_value(i)->user_port );
177 
178  _connections.get_value(i)->user_time = new_time;
179  }
180  _passed_time = 0;
181 
182  return true;
183  }
184  return false;
