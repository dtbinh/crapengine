259 static void makeDistanceMap(const uint8_t* _img, uint8_t* _outImg, uint32_t _width, uint32_t _height)
260 {
261 #if USE_EDTAA3
262  int16_t* xdist = (int16_t*)malloc(_width * _height * sizeof(int16_t) );
263  int16_t* ydist = (int16_t*)malloc(_width * _height * sizeof(int16_t) );
264  double* gx = (double*)calloc(_width * _height, sizeof(double) );
265  double* gy = (double*)calloc(_width * _height, sizeof(double) );
266  double* data = (double*)calloc(_width * _height, sizeof(double) );
267  double* outside = (double*)calloc(_width * _height, sizeof(double) );
268  double* inside = (double*)calloc(_width * _height, sizeof(double) );
269  uint32_t ii;
270 
271  // Convert img into double (data)
272  double img_min = 255, img_max = -255;
273  for (ii = 0; ii < _width * _height; ++ii)
274  {
275  double v = _img[ii];
276  data[ii] = v;
277  if (v > img_max)
278  {
279  img_max = v;
280  }
281 
282  if (v < img_min)
283  {
284  img_min = v;
285  }
286  }
287 
288  // Rescale image levels between 0 and 1
289  for (ii = 0; ii < _width * _height; ++ii)
290  {
291  data[ii] = (_img[ii] - img_min) / (img_max - img_min);
292  }
293 
294  // Compute outside = edtaa3(bitmap); % Transform background (0's)
295  computegradient(data, _width, _height, gx, gy);
296  edtaa3(data, gx, gy, _width, _height, xdist, ydist, outside);
297  for (ii = 0; ii < _width * _height; ++ii)
298  {
299  if (outside[ii] < 0)
300  {
301  outside[ii] = 0.0;
302  }
303  }
304 
305  // Compute inside = edtaa3(1-bitmap); % Transform foreground (1's)
306  memset(gx, 0, sizeof(double) * _width * _height);
307  memset(gy, 0, sizeof(double) * _width * _height);
308  for (ii = 0; ii < _width * _height; ++ii)
309  {
310  data[ii] = 1.0 - data[ii];
311  }
312 
313  computegradient(data, _width, _height, gx, gy);
314  edtaa3(data, gx, gy, _width, _height, xdist, ydist, inside);
315  for (ii = 0; ii < _width * _height; ++ii)
316  {
317  if (inside[ii] < 0)
318  {
319  inside[ii] = 0.0;
320  }
321  }
322 
323  // distmap = outside - inside; % Bipolar distance field
324  uint8_t* out = _outImg;
325  for (ii = 0; ii < _width * _height; ++ii)
326  {
327  outside[ii] -= inside[ii];
328  outside[ii] = 128 + outside[ii] * 16;
329 
330  if (outside[ii] < 0)
331  {
332  outside[ii] = 0;
333  }
334 
335  if (outside[ii] > 255)
336  {
337  outside[ii] = 255;
338  }
339 
340  out[ii] = 255 - (uint8_t) outside[ii];
341  }
342 
343  free(xdist);
344  free(ydist);
345  free(gx);
346  free(gy);
347  free(data);
348  free(outside);
349  free(inside);
350 #else
351  sdfBuild(_outImg, _width, 8.0f, _img, _width, _height, _width);
352 #endif // USE_EDTAA3
353 }

355 bool TrueTypeFont::bakeGlyphDistance(CodePoint _codePoint, GlyphInfo& _glyphInfo, uint8_t* _outBuffer)
356 {
357  BX_CHECK(m_font != NULL, "TrueTypeFont not initialized");
358 
359  _glyphInfo.glyphIndex = FT_Get_Char_Index(m_font->face, _codePoint);
360 
361  FT_Int32 loadMode = FT_LOAD_DEFAULT | FT_LOAD_NO_HINTING;
362  FT_Render_Mode renderMode = FT_RENDER_MODE_NORMAL;
363 
364  FT_GlyphSlot slot = m_font->face->glyph;
365  FT_Error error = FT_Load_Glyph(m_font->face, _glyphInfo.glyphIndex, loadMode);
366  if (error)
367  {
368  return false;
369  }
370 
371  FT_Glyph glyph;
372  error = FT_Get_Glyph(slot, &glyph);
373  if (error)
374  {
375  return false;
376  }
377 
378  error = FT_Glyph_To_Bitmap(&glyph, renderMode, 0, 1);
379  if (error)
380  {
381  return false;
382  }
383 
384  FT_BitmapGlyph bitmap = (FT_BitmapGlyph)glyph;
385 
386  int32_t ww = bitmap->bitmap.width;
387  int32_t hh = bitmap->bitmap.rows;
388 
389  glyphInfoInit(_glyphInfo, bitmap, slot, _outBuffer, 1);
390 
391  FT_Done_Glyph(glyph);
392 
393  if (ww * hh > 0)
394  {
395  uint32_t dw = 6;
396  uint32_t dh = 6;
397 
398  uint32_t nw = ww + dw * 2;
399  uint32_t nh = hh + dh * 2;
400  BX_CHECK(nw * nh < 128 * 128, "Buffer overflow (size %d)", nw * nh);
401 
402  uint32_t buffSize = nw * nh * sizeof(uint8_t);
403 
404  uint8_t* alphaImg = (uint8_t*)malloc(buffSize);
405  memset(alphaImg, 0, nw * nh * sizeof(uint8_t) );
406 
407  //copy the original buffer to the temp one
408  for (uint32_t ii = dh; ii < nh - dh; ++ii)
409  {
410  memcpy(alphaImg + ii * nw + dw, _outBuffer + (ii - dh) * ww, ww);
411  }
412 
413  makeDistanceMap(alphaImg, _outBuffer, nw, nh);
414  free(alphaImg);
415 
416  _glyphInfo.offset_x -= (float)dw;
417  _glyphInfo.offset_y -= (float)dh;
418  _glyphInfo.width = (float)nw;
419  _glyphInfo.height = (float)nh;
420  }
421 
422  return true;
423 }
