113 void calcTangents(void* _vertices, uint16_t _numVertices, bgfx::VertexDecl _decl, const uint16_t* _indices, uint32_t _numIndices)
114 {
115  struct PosTexcoord
116  {
117  float m_x;
118  float m_y;
119  float m_z;
120  float m_pad0;
121  float m_u;
122  float m_v;
123  float m_pad1;
124  float m_pad2;
125  };
126 
127  float* tangents = new float[6*_numVertices];
128  memset(tangents, 0, 6*_numVertices*sizeof(float) );
129 
130  PosTexcoord v0;
131  PosTexcoord v1;
132  PosTexcoord v2;
133 
134  for (uint32_t ii = 0, num = _numIndices/3; ii < num; ++ii)
135  {
136  const uint16_t* indices = &_indices[ii*3];
137  uint32_t i0 = indices[0];
138  uint32_t i1 = indices[1];
139  uint32_t i2 = indices[2];
140 
141  bgfx::vertexUnpack(&v0.m_x, bgfx::Attrib::Position, _decl, _vertices, i0);
142  bgfx::vertexUnpack(&v0.m_u, bgfx::Attrib::TexCoord0, _decl, _vertices, i0);
143 
144  bgfx::vertexUnpack(&v1.m_x, bgfx::Attrib::Position, _decl, _vertices, i1);
145  bgfx::vertexUnpack(&v1.m_u, bgfx::Attrib::TexCoord0, _decl, _vertices, i1);
146 
147  bgfx::vertexUnpack(&v2.m_x, bgfx::Attrib::Position, _decl, _vertices, i2);
148  bgfx::vertexUnpack(&v2.m_u, bgfx::Attrib::TexCoord0, _decl, _vertices, i2);
149 
150  const float bax = v1.m_x - v0.m_x;
151  const float bay = v1.m_y - v0.m_y;
152  const float baz = v1.m_z - v0.m_z;
153  const float bau = v1.m_u - v0.m_u;
154  const float bav = v1.m_v - v0.m_v;
155 
156  const float cax = v2.m_x - v0.m_x;
157  const float cay = v2.m_y - v0.m_y;
158  const float caz = v2.m_z - v0.m_z;
159  const float cau = v2.m_u - v0.m_u;
160  const float cav = v2.m_v - v0.m_v;
161 
162  const float det = (bau * cav - bav * cau);
163  const float invDet = 1.0f / det;
164 
165  const float tx = (bax * cav - cax * bav) * invDet;
166  const float ty = (bay * cav - cay * bav) * invDet;
167  const float tz = (baz * cav - caz * bav) * invDet;
168 
169  const float bx = (cax * bau - bax * cau) * invDet;
170  const float by = (cay * bau - bay * cau) * invDet;
171  const float bz = (caz * bau - baz * cau) * invDet;
172 
173  for (uint32_t jj = 0; jj < 3; ++jj)
174  {
175  float* tanu = &tangents[indices[jj]*6];
176  float* tanv = &tanu[3];
177  tanu[0] += tx;
178  tanu[1] += ty;
179  tanu[2] += tz;
180 
181  tanv[0] += bx;
182  tanv[1] += by;
183  tanv[2] += bz;
184  }
185  }
186 
187  for (uint32_t ii = 0; ii < _numVertices; ++ii)
188  {
189  const float* tanu = &tangents[ii*6];
190  const float* tanv = &tangents[ii*6 + 3];
191 
192  float normal[4];
193  bgfx::vertexUnpack(normal, bgfx::Attrib::Normal, _decl, _vertices, ii);
194  float ndt = bx::vec3Dot(normal, tanu);
195 
196  float nxt[3];
197  bx::vec3Cross(nxt, normal, tanu);
198 
199  float tmp[3];
200  tmp[0] = tanu[0] - normal[0] * ndt;
201  tmp[1] = tanu[1] - normal[1] * ndt;
202  tmp[2] = tanu[2] - normal[2] * ndt;
203 
204  float tangent[4];
205  bx::vec3Norm(tangent, tmp);
206 
207  tangent[3] = bx::vec3Dot(nxt, tanv) < 0.0f ? -1.0f : 1.0f;
208  bgfx::vertexPack(tangent, true, bgfx::Attrib::Tangent, _decl, _vertices, ii);
209  }
210 
211  delete [] tangents;
212 }
