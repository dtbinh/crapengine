149  void run()
150  {
151  const char* argv[1] = { "sdl.so" };
152  m_mte.m_argc = 1;
153  m_mte.m_argv = const_cast<char**>(argv);
154 
155  SDL_Init(SDL_INIT_VIDEO);
156 
157  m_window = SDL_CreateWindow("bgfx"
158  , SDL_WINDOWPOS_UNDEFINED
159  , SDL_WINDOWPOS_UNDEFINED
160  , m_width
161  , m_height
162  , SDL_WINDOW_SHOWN
163  | SDL_WINDOW_OPENGL
164  | SDL_WINDOW_RESIZABLE
165  );
166 
167  s_userEventStart = SDL_RegisterEvents(3);
168 
169  bgfx::sdlSetWindow(m_window);
170  m_thread.init(MainThreadEntry::threadFunc, &m_mte);
171 
172  // Force window resolution...
173  setWindowSize(m_width, m_height, true);
174 
175  bool exit = false;
176  SDL_Event event;
177  while (!exit && SDL_WaitEvent(&event) )
178  {
179  switch (event.type)
180  {
181  case SDL_QUIT:
182  m_eventQueue.postExitEvent();
183  exit = true;
184  break;
185 
186  case SDL_MOUSEMOTION:
187  {
188  const SDL_MouseMotionEvent& mev = event.motion;
189  m_eventQueue.postMouseEvent(mev.x, mev.y);
190  }
191  break;
192 
193  case SDL_MOUSEBUTTONDOWN:
194  case SDL_MOUSEBUTTONUP:
195  {
196  const SDL_MouseButtonEvent& mev = event.button;
197  m_eventQueue.postMouseEvent(mev.x, mev.y, MouseButton::Left, mev.type == SDL_MOUSEBUTTONDOWN);
198  }
199  break;
200 
201  case SDL_KEYDOWN:
202  case SDL_KEYUP:
203  {
204  const SDL_KeyboardEvent& kev = event.key;
205  uint8_t modifiers = translateKeyModifiers(kev.keysym.mod);
206  Key::Enum key = translateKey(kev.keysym.scancode);
207  m_eventQueue.postKeyEvent(key, modifiers, kev.state == SDL_PRESSED);
208  }
209  break;
210 
211  case SDL_WINDOWEVENT:
212  {
213  const SDL_WindowEvent& wev = event.window;
214  switch (wev.event)
215  {
216  case SDL_WINDOWEVENT_RESIZED:
217  case SDL_WINDOWEVENT_SIZE_CHANGED:
218  setWindowSize(wev.data1, wev.data2);
219  break;
220 
221  case SDL_WINDOWEVENT_SHOWN:
222  case SDL_WINDOWEVENT_HIDDEN:
223  case SDL_WINDOWEVENT_EXPOSED:
224  case SDL_WINDOWEVENT_MOVED:
225  case SDL_WINDOWEVENT_MINIMIZED:
226  case SDL_WINDOWEVENT_MAXIMIZED:
227  case SDL_WINDOWEVENT_RESTORED:
228  case SDL_WINDOWEVENT_ENTER:
229  case SDL_WINDOWEVENT_LEAVE:
230  case SDL_WINDOWEVENT_FOCUS_GAINED:
231  case SDL_WINDOWEVENT_FOCUS_LOST:
232  break;
233 
234  case SDL_WINDOWEVENT_CLOSE:
235  m_eventQueue.postExitEvent();
236  exit = true;
237  break;
238  }
239  }
240  break;
241 
242  default:
243  {
244  const SDL_UserEvent uev = event.user;
245  if (SDL_USER_SET_WINDOW_SIZE == event.type)
246  {
247  uint32_t width = *(uint32_t*)&uev.data1;
248  uint32_t height = *(uint32_t*)&uev.data2;
249  setWindowSize(width, height);
250  }
251  else if (SDL_USER_TOGGLE_WINDOW_FRAME == event.type)
252  {
253  DBG("SDL_USER_TOGGLE_WINDOW_FRAME");
254  }
255  else if (SDL_USER_MOUSE_LOCK == event.type)
256  {
257  setMouseLock(!!uev.code);
258  }
259  }
260  break;
261  }
262  }
