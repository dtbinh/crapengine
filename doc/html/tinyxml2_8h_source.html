345  virtual void* Alloc() {
346  if ( !_root ) {
347  // Need a new block.
348  Block* block = new Block();
349  _blockPtrs.Push( block );
350 
351  for( int i=0; i<COUNT-1; ++i ) {
352  block->chunk[i].next = &block->chunk[i+1];
353  }
354  block->chunk[COUNT-1].next = 0;
355  _root = block->chunk;
356  }
357  void* result = _root;
358  _root = _root->next;
359 
360  ++_currentAllocs;
361  if ( _currentAllocs > _maxAllocs ) {
362  _maxAllocs = _currentAllocs;
363  }
364  _nAllocs++;
365  _nUntracked++;
366  return result;
367  }
368  virtual void Free( void* mem ) {
369  if ( !mem ) {
370  return;
371  }

373  Chunk* chunk = static_cast<Chunk*>( mem );
374 #ifdef DEBUG
375  memset( chunk, 0xfe, sizeof(Chunk) );
376 #endif
377  chunk->next = _root;
378  _root = chunk;
379  }

383  }
