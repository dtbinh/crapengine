193 int32_t RectanglePacker::fit(uint32_t _skylineNodeIndex, uint16_t _width, uint16_t _height)
194 {
195  int32_t width = _width;
196  int32_t height = _height;
197 
198  const Node& baseNode = m_skyline[_skylineNodeIndex];
199 
200  int32_t xx = baseNode.x, yy;
201  int32_t widthLeft = width;
202  int32_t ii = _skylineNodeIndex;
203 
204  if ( (xx + width) > (int32_t)(m_width - 1) )
205  {
206  return -1;
207  }
208 
209  yy = baseNode.y;
210  while (widthLeft > 0)
211  {
212  const Node& node = m_skyline[ii];
213  if (node.y > yy)
214  {
215  yy = node.y;
216  }
217 
218  if ( (yy + height) > (int32_t)(m_height - 1) )
219  {
220  return -1;
221  }
222 
223  widthLeft -= node.width;
224  ++ii;
225  }
226 
227  return yy;
228 }
