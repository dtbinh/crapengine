104 pointer_t<void> general_allocator::allocate( uint32_t size, uint32_t alignment, uint32_t offset )
105 {
106  const uint32_t total_size = size + alignment + offset + sizeof(memory_block);
107  pointer_t<memory_block> current_block = _block_start;
108 
109  while(current_block.as_uint8_t < _block_end.as_uint8_t )
110  {
111  memory_block& current = *current_block.as_type;
112  const uint32_t free_memory = (current.offset_next - current.used_memory) - sizeof(memory_block);
113 
114  if( free_memory >= total_size )
115  {
116  pointer_t<memory_block> next_block = current_block.as_uint8_t + current.offset_next;
117  pointer_t<memory_block> temp_block = current_block.as_type;
118 
119  temp_block.as_uint8_t += current.used_memory + offset + (sizeof(memory_block)*2);
120  temp_block.align_forward(alignment);
121  temp_block.as_uint8_t -= ( offset + sizeof(memory_block) );
122 
123  memory_block& inserted = *temp_block.as_type;
124 
125  inserted.offset_prev = temp_block.as_uint8_t - current_block.as_uint8_t;
126  inserted.offset_next = next_block.as_uint8_t - temp_block.as_uint8_t;
127  inserted.used_memory = (size+offset);
128 
129  memory_block& next = *next_block.as_type;
130 
131  current.offset_next = inserted.offset_prev;
132  next.offset_prev = inserted.offset_next;
133 
134  temp_block.as_type += 1;
135  return temp_block.as_void;
136  }
137 
138  current_block.as_uint8_t += current.offset_next;
139  }
140 
141  return 0;
